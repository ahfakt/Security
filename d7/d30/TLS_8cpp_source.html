115TLSEncrypt::writeBytes(std::byte const* src, std::size_t size)
116{

118 int r = SSL_is_init_finished(mSSL)
119 ? SSL_write_ex(mSSL, src, size, &inl)
120 : SSL_do_handshake(mSSL);
121 if (r == 1) {
122 if (inl)

124 return inl;

139bool
140TLSEncrypt::shutdown()
141{
142 int r = SSL_shutdown(mSSL);
143 if (r == 1)
144 return true;
145 if (r == 0) {
146 sendData();
147 flush();
148 try {
149 wantRecvData();

153 return shutdown();
154 }
